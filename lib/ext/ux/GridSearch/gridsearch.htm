  264. ,commitChanges:function() {
  265. var records = this.store.getModifiedRecords();
  266. if(!records.length) {
  267. return;
  268. }
  269. var data = [];
  270. Ext.each(records, function(r, i) {
  271. data.push(r.data);
  272. }, this);
  273. var o = {
  274. url:this.url
  275. ,method:'post'
  276. ,callback:this.requestCallback
  277. ,scope:this
  278. ,params:{
  279. cmd:'saveData'
  280. ,objName:this.objName
  281. ,data:Ext.encode(data)
  282. }
  283. };
  284. Ext.Ajax.request(o);
  285. } // eo function commitChanges
  286. // }}}
  287. // {{{
  288. ,requestCallback:function(options, success, response) {
  289. if(true !== success) {
  290. this.showError(response.responseText);
  291. return;
  292. }
  293. try {
  294. var o = Ext.decode(response.responseText);
  295. }
  296. catch(e) {
  297. this.showError(response.responseText, 'Cannot decode JSON object');
  298. return;
  299. }
  300. if(true !== o.success) {
  301. this.showError(o.error || 'Unknown error');
  302. return;
  303. }
  304.  
  305. switch(options.params.cmd) {
  306. case 'saveData':
  307. var records = this.store.getModifiedRecords();
  308. Ext.each(records, function(r, i) {
  309. if(o.insertIds && o.insertIds[i]) {
  310. r.set(this.idName, o.insertIds[i]);
  311. delete(r.data.newRecord);
  312. }
  313. });
  314. this.store.commitChanges();
  315. break;
  316.  
  317. case 'deleteData':
  318. break;
  319. }
  320. } // eo function requestCallback
