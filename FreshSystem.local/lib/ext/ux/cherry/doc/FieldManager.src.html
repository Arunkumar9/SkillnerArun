  5 /** Creates a new FieldManager class in the Netbox Filter Framework, eventually instantiating the fields.<br>
  6   * <b> NB: </b> If availableValues is specified and the type of the field is string the STRING_LIST and STRING_NOT_IN_LIST operators will be added to the fields.
  7   * @constructor
  8   * @param (Object) config The configuration options for this object. Optional, if not present an empty FieldManager is created.
  9   * @config {String} id The id of the field (for example the id of the column)
 10   * @config {String} label The label of the field (for example the header of the column)
 11   * @config {Array} defaultValues The default values of the field in the format {value: ... , label: ...} Optional
 12   * @config {String} type The type of the field. Available values are: string, enum, float, int, date. Default type is string
 13   * @config {String} format Only for dates, the format of the date. Look Ext.ux.netbox.data.DateFilterType for more details
 14   * @config {Ext.data.Store} availableValues The store containing the available values for this field. Look at Ext.ux.netbox.core.Field.setAvailableValues for more details
 15   * @config {boolean} remoteStore True if the store is remote, false otherwise. Look at Ext.ux.netbox.core.Field.setStoreRemote for more details
 16   * @config {boolean} forceReload True if you want reload the store everytime expand the combo. Look at Ext.ux.netbox.core.Field.setForceReload for more details
 17   * @config {function} validate The function used to validate the values of the Field. Look at Ext.ux.netbox.core.Field.setValidateFn for more details
 18   * @class This class manages the filter fields. You should use an instance of this class (obtained using filterModel.getFieldManager) to add or remove filter fields, or to obtain them to add or remove operators.
 19   * More than one filter model can have the same FieldManager, allowing to have different sets of filters, but on the same set of fields.
 20   */
 21 Ext.ux.netbox.core.FieldManager=function(config){
 22   Ext.ux.netbox.core.FieldManager.superclass.constructor.call(this);
 23   this.addEvents(/** @scope Ext.ux.netbox.core.FieldManager.prototype */{
 24     /** Fires when a Field is added
 25       * @event fieldAdded
 26       * @param {Ext.ux.netbox.core.Field} field The added field
 27       */
 28     fieldAdded : true,
 29     /** Fires when a Field is removed
 30       * @event fieldRemoved
 31       * @param {Ext.ux.netbox.core.Field} field The removed field
 32       */
 33     fieldRemoved : true
 34   });
 35   /**
 36     * @property {Ext.util.MixedCollection} fields The list of all fields for this FieldManager
 37     * @private
 38     */
 39   this.fields=new Ext.util.MixedCollection(false,function(field){return(field.getId())});
 40   if(config!==undefined){
 41     for(var i=0; i< config.length; i++){
 42       var field;
 43       if(config[i].type===undefined){
 44         config[i].type="string";
 45       }
 46 
 47       switch(config[i].type){
 48         case "string":
 49           field=new Ext.ux.netbox.string.StringField(config[i].id,config[i].label);
 50           break;
 51         case "enum":
 52           field=new Ext.ux.netbox.string.EnumField(config[i].id,config[i].label);
 53           break;
 54         case "float":
 55         case "int":
 56           field=new Ext.ux.netbox.number.NumberField(config[i].id,config[i].label);
 57           break;
 58         case "date":
 59           field=new Ext.ux.netbox.date.DateField(config[i].id,config[i].label,config[i].format);
 60           break;
 61         default:
 62           continue;
 63       }
 64       if(config[i].availableValues!==undefined){
 65         field.setAvailableValues(config[i].availableValues);
 66         if(config[i].remoteStore!==undefined){
 67           field.setStoreRemote(config[i].remoteStore);
 68         }
 69         if(config[i].type=="string" || config[i].type=="enum"){
 70           field.addOperator(new Ext.ux.netbox.string.StringListOperator('STRING_LIST',field.stringListText));
 71           field.addOperator(new Ext.ux.netbox.string.StringListOperator('STRING_NOT_IN_LIST',field.stringNotListText));
 72         }
 73       }
 74       if(config[i].defaultValues!==undefined)
 75         field.setDefaultValues(config[i].defaultValues);
 76       if(config[i].forceReload!==undefined)
 77         field.setForceReload(config[i].forceReload);
 78       if(config[i].validate!==undefined)
 79         field.setValidateFn(config[i].validate);
 80 
 81       this.addField(field);
 82     }
 83   }
 84 }
 85 
 86 Ext.extend(Ext.ux.netbox.core.FieldManager,Ext.util.Observable,/** @scope Ext.ux.netbox.core.FieldManager.prototype */
 87 {
 88   /** This method returns the list of all fields.
 89     * @return {Array of Ext.ux.netbox.core.Field} The collection of fields for this Filter
 90     */
 91   getAllFields : function(){
 92     return(this.fields.getRange());
 93   },
 94   /** This method returns a field by Id or null if it is not found.
 95     * @param {String} id The id of field
 96     * @return {Ext.ux.netbox.core.Field} The field with the given id
 97     */
 98   getFieldById : function(id){
 99     var field=this.fields.get(id);
100     if(!field){
101       return(null);
102     }
103     return(field);
104   },
105   /** This method add a Field to the array in FieldManager.
106     * If this succeeds the event "fieldAdded" is triggered.
107     * @param {Ext.ux.netbox.core.Field} field The Field to add
108     */
109   addField : function(field){
110     this.fields.add(field);
111     this.fireEvent("fieldAdded",field);
112   },
113   /** This method remove a Field from the array in FieldManager.
114     * If this succeeds the event "fieldRemoved" is triggered.
115     * @param {Ext.ux.netbox.core.Field} field The Field to remove
116     */
117   removeField : function(field){
118     if(this.fields.removeKey(field.getId()))
119     this.fireEvent("fieldRemoved",field);
120   }
121 
122 });
